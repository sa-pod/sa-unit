Go

获取库

go get github.com/sa-pod/sa-unit

使用

package main
import (
    "fmt"

    "github.com/sa-pod/sa-unit"
)
func main() {
    kg := "18kg"
    jin,unit := unit.To(kg, "斤")
    fmt.Println(jin, unit)
    g := "20g=>斤"
    jin, unit = unit.To(g)
    fmt.Println(jin, unit)
    n,nu = 20, "kg"
    jin, unit = unit.To(n, nu,"斤")
    fmt.Println(jin, unit)
}
